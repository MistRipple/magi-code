## 阶段 2: ConflictResolver 冲突解决机制 ✅

### 问题分析
- **优先级不明**: 用户偏好 vs 画像推荐 vs 执行统计的选择顺序不清晰
- **决策不透明**: 无法追踪为什么选择了特定 CLI
- **无法定制**: 无法根据场景调整冲突解决策略

### 解决方案
```typescript
// src/task/conflict-resolver.ts
export class ConflictResolver {
  /**
   * 4 级优先级系统:
   * Level 1: 用户手动选择 (always-respect)
   * Level 2: 执行统计推荐 (健康度阈值 >= 0.7)
   * Level 3: 画像配置推荐
   * Level 4: 硬编码默认值 (fallback)
   */
  resolve(options: ConflictOptions): ConflictResolution {
    if (options.userPreference) {
      return { cli: options.userPreference, level: 'user', reason: '用户指定' };
    }

    if (options.statsRecommendation && this.isHealthy(options.statsRecommendation)) {
      return { cli: options.statsRecommendation, level: 'stats', reason: '统计推荐' };
    }

    if (options.profileRecommendation) {
      return { cli: options.profileRecommendation, level: 'profile', reason: '画像推荐' };
    }

    return { cli: options.availableClis[0], level: 'default', reason: '默认回退' };
  }
}
```

## 阶段 3: TaskDependencyGraph 文件依赖调度 ✅

### 问题分析
- **文件冲突**: 多个任务同时修改同一文件可能导致冲突
- **无序执行**: 并行任务可能相互干扰
- **调度不智能**: 无法根据文件依赖优化执行顺序

### 解决方案
```typescript
// src/orchestrator/task-dependency-graph.ts
export class TaskDependencyGraph {
  /**
   * 文件依赖管理:
   * 1. addTask() 时记录 targetFiles
   * 2. detectFileConflicts() 检测冲突
   * 3. addFileDependencies() 自动添加依赖关系
   */
  addTask(id: string, description: string, dependencies: {}, targetFiles: string[]) {
    this.nodes.set(id, {
      id,
      description,
      dependencies: new Set(Object.keys(dependencies)),
      targetFiles,
    });
  }

  detectFileConflicts(): FileConflict[] {
    const fileMap = new Map<string, string[]>();
    for (const [id, node] of this.nodes) {
      for (const file of node.targetFiles) {
        if (!fileMap.has(file)) fileMap.set(file, []);
        fileMap.get(file)!.push(id);
      }
    }
    return Array.from(fileMap.entries())
      .filter(([_, tasks]) => tasks.length > 1)
      .map(([file, tasks]) => ({ file, tasks }));
  }

  addFileDependencies(strategy: 'sequential' | 'parallel'): number {
    const conflicts = this.detectFileConflicts();
    let added = 0;

    if (strategy === 'sequential') {
      for (const { file, tasks } of conflicts) {
        const sorted = tasks.sort();
        for (let i = 1; i < sorted.length; i++) {
          this.addDependency(sorted[i], sorted[i - 1]);
          added++;
        }
      }
    }

    return added;
  }
}
```

## 阶段 4: MessageDeduplicator 消息去重 ✅

### 问题分析
- **消息重复**: 同一消息可能被多次发送到 UI
- **UI 覆盖**: 对话面板消息相互覆盖
- **进度混乱**: STREAMING 消息过于频繁导致 UI 卡顿

### 解决方案
```typescript
// src/normalizer/message-deduplicator.ts
export class MessageDeduplicator {
  private messages = new Map<string, MessageState>();

  shouldSend(message: StandardMessage): boolean {
    const key = this.getMessageKey(message);
    const state = this.messages.get(key);

    // 规则 1: STARTED 消息总是发送
    if (message.lifecycle === MessageLifecycle.STARTED) {
      this.messages.set(key, {
        id: message.id,
        lifecycle: message.lifecycle,
        lastSentAt: Date.now(),
      });
      return true;
    }

    // 规则 2: 已完成消息不再发送
    if (state?.lifecycle === MessageLifecycle.COMPLETED) {
      this.debug(`[MessageDeduplicator] 消息已完成，跳过: ${message.id}`);
      return false;
    }

    // 规则 3: STREAMING 消息限制间隔
    if (message.lifecycle === MessageLifecycle.STREAMING) {
      const now = Date.now();
      const elapsed = state?.lastSentAt ? now - state.lastSentAt : Infinity;

      if (elapsed < this.config.minStreamInterval) {
        return false; // 间隔太短,跳过
      }

      this.messages.set(key, { ...state!, lastSentAt: now });
      return true;
    }

    // 规则 4: COMPLETED 消息标记完成
    if (message.lifecycle === MessageLifecycle.COMPLETED) {
      this.messages.set(key, {
        id: message.id,
        lifecycle: MessageLifecycle.COMPLETED,
        lastSentAt: Date.now(),
      });
      return true;
    }

    return true;
  }

  private getMessageKey(message: StandardMessage): string {
    return `${message.source}-${message.id}`;
  }
}
```
