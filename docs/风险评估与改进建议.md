## 1. 架构设计风险

### 🔴 1.1 单点故障风险

**问题描述**:
- **OrchestratorAgent 是单点**: 如果编排者崩溃，整个任务失败，没有备份机制
- **Claude 强依赖**: 任务分析和汇总都依赖 Claude，如果 Claude 不可用，系统完全无法工作
- **消息总线单点**: MessageBus 如果出问题，Worker 之间无法通信

**代码位置**:
```typescript
// src/orchestrator/intelligent-orchestrator.ts
// 只有一个 orchestratorAgent 实例，没有备份
this.orchestratorAgent = new OrchestratorAgent(...);
```

**风险影响**:
- 系统可用性低
- 无法处理 Claude API 故障
- 用户任务可能丢失

**改进建议**:

```typescript
// 1. 添加降级策略
class IntelligentOrchestrator {
    private fallbackStrategies = {
        'claude_unavailable': async () => {
            // 降级到 Codex 或 Gemini 进行分析
            return await this.analyzeWithFallback('codex');
        },
        'orchestrator_crash': async () => {
            // 从快照恢复状态
            return await this.recoverFromSnapshot();
        }
    };
}

// 2. 添加健康检查
async checkHealth(): Promise<HealthStatus> {
    return {
        claude: await this.cliFactory.checkCLIHealth('claude'),

        messageBus: this.messageBus.isHealthy()
    };
}
```

```typescript
enum ErrorType {
    NETWORK_ERROR = 'network',      // 可重试

    PERMISSION_ERROR = 'permission', // 不可重试

class RetryStrategy {
    shouldRetry(error: Error, attempt: number): boolean {
        const errorType = this.classifyError(error);
        
        switch (errorType) {
            case ErrorType.NETWORK_ERROR:
            case ErrorType.TIMEOUT_ERROR:
```

```typescript
// src/orchestrator/file-lock-manager.ts
class FileLockManager {
    private locks = new Map<string, {workerId: string, timestamp: number}>();

    async acquireLock(filePath: string, workerId: string, timeout: number = 30000): Promise<boolean> {
        const existing = this.locks.get(filePath);

        // 检查是否已被锁定
        if (existing) {
            // 检查锁是否过期
            if (Date.now() - existing.timestamp > timeout) {
                console.warn(`[FileLock] 锁已过期，强制释放: ${filePath}`);
                this.locks.delete(filePath);
            } else {
                console.warn(`[FileLock] 文件 ${filePath} 已被 ${existing.workerId} 锁定`);
                return false;
            }
        }

        this.locks.set(filePath, { workerId, timestamp: Date.now() });
        return true;
    }

    releaseLock(filePath: string, workerId: string): void {
        const existing = this.locks.get(filePath);
        if (existing && existing.workerId === workerId) {
            this.locks.delete(filePath);
        }
    }
}

// 2. 执行前检查文件冲突
private async checkFileConflicts(subTasks: SubTask[]): Promise<ConflictReport> {
    const fileMap = new Map<string, string[]>();

    for (const task of subTasks) {
        for (const file of task.targetFiles || []) {
            if (!fileMap.has(file)) {
                fileMap.set(file, []);
            }
            fileMap.get(file)!.push(task.id);
        }
    }

    const conflicts: Array<{file: string, tasks: string[]}> = [];
    for (const [file, tasks] of fileMap.entries()) {
        if (tasks.length > 1) {
            conflicts.push({ file, tasks });
        }
    }

    return { hasConflict: conflicts.length > 0, conflicts };
}
```

```typescript
// 1. 添加清理机制
class BaseNormalizer extends EventEmitter {
    protected activeContexts = new Map<string, ParseContext>();
    private cleanupTimer: NodeJS.Timeout;

    constructor(config: NormalizerConfig) {
        super();
        this.cleanupTimer = setInterval(() => {
            this.cleanupStaleContexts();
        }, 60000);  // 每分钟清理一次
    }

    private cleanupStaleContexts(): void {
        const now = Date.now();
        const staleThreshold = 300000;  // 5分钟

        for (const [id, context] of this.activeContexts.entries()) {
            if (now - context.startTime > staleThreshold) {
                console.warn(`[Normalizer] 清理过期上下文: ${id}`);
                this.activeContexts.delete(id);
            }
        }
    }

    dispose(): void {
        clearInterval(this.cleanupTimer);
        this.activeContexts.clear();
        this.removeAllListeners();
    }
}
```

```typescript
class InputSanitizer {
    sanitizeUserPrompt(prompt: string): string {
        let sanitized = prompt
            .replace(/```[\s\S]*?```/g, '[代码块已移除]')
            .replace(/<\|.*?\|>/g, '')
            .replace(/\[SYSTEM\]/gi, '')
            .trim();

        if (sanitized.length > 10000) {
            sanitized = sanitized.substring(0, 10000) + '... [已截断]';
        }

        return sanitized;

    validateFilePath(filePath: string, workspaceRoot: string): boolean {
        const normalized = path.normalize(filePath);
        const resolved = path.resolve(workspaceRoot, normalized);

        if (!resolved.startsWith(workspaceRoot)) {
            throw new Error(`非法文件路径: ${filePath}`);
        }

        const sensitivePatterns = [/\.env$/, /\.git\//, /\.ssh\//];
        for (const pattern of sensitivePatterns) {
            if (pattern.test(normalized)) {
                throw new Error(`禁止访问敏感文件: ${filePath}`);
            }
        }

        return true;
    }
}
```

**问题描述**:
- **硬编码规则**: CLI 分配规则写死在 Prompt 中
- **无学习能力**: 不会根据历史执行结果优化分配策略

```typescript
// 1. 添加 CLI 性能追踪
class CLIPerformanceTracker {
    private stats = new Map<CLIType, {
        successCount: number;
        failureCount: number;
        avgDuration: number;
        taskTypes: Map<string, number>;  // 任务类型 -> 成功次数
    }>();

    recordExecution(cli: CLIType, taskType: string, success: boolean, duration: number) {
        const stat = this.stats.get(cli) || {
            successCount: 0,
            failureCount: 0,
            avgDuration: 0,
            taskTypes: new Map()
        };

        if (success) {
            stat.successCount++;
            stat.taskTypes.set(taskType, (stat.taskTypes.get(taskType) || 0) + 1);
        } else {
            stat.failureCount++;
        }

        stat.avgDuration = (stat.avgDuration * (stat.successCount + stat.failureCount - 1) + duration)
                          / (stat.successCount + stat.failureCount);

        this.stats.set(cli, stat);
    }

    getBestCLIForTask(taskType: string): CLIType {
        let bestCLI: CLIType = 'claude';
        let bestScore = 0;

        for (const [cli, stat] of this.stats.entries()) {
            const successRate = stat.successCount / (stat.successCount + stat.failureCount);
            const taskSuccess = stat.taskTypes.get(taskType) || 0;
            const score = successRate * 0.7 + (taskSuccess / stat.successCount) * 0.3;

            if (score > bestScore) {
                bestScore = score;
                bestCLI = cli;
            }
        }

        return bestCLI;
    }
}

// 2. 动态 CLI 切换
class CLIFallbackManager {
    async executeWithFallback(
        primaryCLI: CLIType,
        task: SubTask,
        context?: string
    ): Promise<ExecutionResult> {
        const fallbackOrder: CLIType[] = this.getFallbackOrder(primaryCLI);

        for (const cli of fallbackOrder) {
            try {
                const result = await this.executeTask(cli, task, context);
                if (result.success) {
                    if (cli !== primaryCLI) {
                        console.log(`[Fallback] 使用 ${cli} 替代 ${primaryCLI} 成功`);
                    }
                    return result;
                }
            } catch (error) {
                console.warn(`[Fallback] ${cli} 执行失败，尝试下一个...`);
            }
        }

        throw new Error('所有 CLI 都执行失败');
    }

    private getFallbackOrder(primaryCLI: CLIType): CLIType[] {
        const allCLIs: CLIType[] = ['claude', 'codex', 'gemini'];
        return [primaryCLI, ...allCLIs.filter(c => c !== primaryCLI)];
    }
}
```

```typescript
        // 记录错误
        this.logError(error, context);

        // 分类错误
        const errorType = this.classifyError(error);

        // 尝试恢复
        if (errorType.recoverable) {
            const recovered = await this.attemptRecovery(error, context);
            if (recovered) {
                return { recovered: true, result: recovered };
            }
        }

        // 通知用户
        this.notifyUser(error, context);

        return { recovered: false, error };
    }
}
```

```typescript
class Logger {
    private level: LogLevel = LogLevel.INFO;
    private logs: LogEntry[] = [];

    log(level: LogLevel, message: string, context?: any) {

            level: LogLevel[level],
```
