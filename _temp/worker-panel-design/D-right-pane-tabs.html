§6 状态机 & 数据契约 right-pane.svelte.ts 

// web/src/stores/right-pane.svelte.ts

type TabKind = 'agent' | 'code';  // v1 只两种；TabKind 是 union，后续扩展加 case

type Tab = {
  id: string;            // agent-tab: workerTabId · code-tab: file path
  kind: TabKind;
  label: string;         // 显示文字
  accentToken: string;   // CSS var name: --color-claude / --info
  payload: AgentPayload | CodePayload;
  lastActivatedAt: number;  // LRU 淘汰用
};

type SessionPaneState = {
  openTabs: Tab[];      // 上限 6，超额 LRU
  activeTabId: string | null;
  collapsed: boolean;    // 折叠 ≠ 销毁 tabs
};

export const rightPaneState = $state<{
  perSession: Map<string, SessionPaneState>;
}>({ perSession: new Map() });

// === 操作 ===
export function openAgentTab(workerTabId: string): void;
export function openCodeTab(params: { path: string; diffHint?: DiffHint; lineRange?: Range }): void;
export function closeTab(tabId: string): void;       // 真销毁；length 归零自动 collapsed=true
export function activateTab(tabId: string): void;
export function togglePane(): void;                // 仅切 collapsed，不动 openTabs
§6 关键点
  • collapsed 与 openTabs 正交：折叠时 openTabs 完整保留
  • 关闭单 tab：从 openTabs 删除；若 openTabs.length === 0 → 强制 collapsed = true
  • 展开按钮触发 togglePane()；若 openTabs 为空则展开为空白 Pane（§5.1）
  • 状态按 sessionId 隔离，切 session = 切 perSession 的 entry，跨会话残留问题天然解决
  • v1 之后扩展浏览器 / 侧边聊天：只需在 TabKind union 加 case，store API 不变
